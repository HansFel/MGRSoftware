Setup-Anleitung für Standby-Server

# Ubuntu/Debian
sudo apt update
sudo apt install postgresql-16 postgresql-client-16

# Service stoppen (wird manuell konfiguriert)
sudo systemctl stop postgresql

# postgresql.conf bearbeiten
sudo nano /etc/postgresql/16/main/postgresql.conf

# Folgende Zeilen hinzufügen/ändern:
wal_level = replica
max_wal_senders = 3
max_replication_slots = 3
hot_standby = on

# pg_hba.conf bearbeiten
sudo nano /etc/postgresql/16/main/pg_hba.conf

# Zeile hinzufügen (IP des Standby-Servers anpassen):
host    replication     replicator      {{ config.standby_host if config and config.standby_host else 'STANDBY_IP' }}/32         scram-sha-256

# Replication-User erstellen
sudo -u postgres psql -c "CREATE USER replicator WITH REPLICATION ENCRYPTED PASSWORD 'sicheres_passwort';"

# PostgreSQL neu starten
sudo systemctl restart postgresql

# Auf dem STANDBY-Server ausführen:

# Datenverzeichnis leeren
sudo systemctl stop postgresql
sudo rm -rf /var/lib/postgresql/16/main/*

# Base Backup vom Primary holen
sudo -u postgres pg_basebackup -h {{ pg_host }} -U replicator -D /var/lib/postgresql/16/main -Fp -Xs -P -R

# standby.signal erstellen (falls nicht vorhanden)
sudo touch /var/lib/postgresql/16/main/standby.signal
sudo chown postgres:postgres /var/lib/postgresql/16/main/standby.signal

# PostgreSQL starten
sudo systemctl start postgresql

# Auf dem PRIMARY-Server:
sudo -u postgres psql -c "SELECT * FROM pg_stat_replication;"

# Auf dem STANDBY-Server:
sudo -u postgres psql -c "SELECT pg_is_in_recovery();"
# Sollte 't' (true) zurückgeben

# Log überwachen:
sudo tail -f /var/log/postgresql/postgresql-16-main.log

# NUR im Notfall auf dem STANDBY-Server ausführen:

# Standby zum Primary promoten
sudo -u postgres pg_ctl promote -D /var/lib/postgresql/16/main

# Oder mit pg_ctlcluster
sudo pg_ctlcluster 16 main promote

# Dann Anwendung auf neuen Server umstellen!
Achtung: Nach einem Failover muss der alte Primary-Server neu als Standby konfiguriert werden!
