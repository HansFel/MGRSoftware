Portainer:

# 1. Alten Portainer stoppen und entfernen
docker stop portainer
docker rm portainer

# 2. Optional: Alte Daten löschen (für kompletten Neustart)
docker volume rm portainer_data
# ODER Daten behalten, um Einstellungen zu übernehmen

# 3. Portainer neu installieren
docker run -d \
  --name portainer \
  --restart=always \
  -p 8000:8000 \
  -p 9443:9443 \
  -v /var/run/docker.sock:/var/run/docker.sock \
  -v portainer_data:/data \
  portainer/portainer-ce:latest

# 4. Portainer ist jetzt erreichbar unter:
# https://raspberry-ip:9443
# z.B. https://192.168.178.36:9443
